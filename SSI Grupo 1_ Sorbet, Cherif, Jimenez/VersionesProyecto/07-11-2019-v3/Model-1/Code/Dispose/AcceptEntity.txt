IF Entity.Corto = TRUE
	IF Entity.tServicio <= 12
		Model.numProgramasCortosOK := Model.numProgramasCortosOK + 1;
	ELSE
		Model.numProgramasCortosBAD := Model.numProgramasCortosBAD + 1;
	END IF;
ELSE 
	IF Entity.tServicio <= 36
		Model.numProgramasLargosOK := Model.numProgramasLargosOK + 1;
	ELSE
		Model.numProgramasLargosBAD := Model.numProgramasLargosBAD + 1;
	END IF;
END IF;
